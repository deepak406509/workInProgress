// Get the parent element of the header title const parentElem = document.querySelector('#header-title').parentElement; // Create a new text node containing "Hello " const helloText = document.createTextNode('Hello '); // Create a new span element to wrap around the header title const spanElem = document.createElement('span'); // Copy the existing header title element and its contents const headerTitleElem = document.querySelector('#header-title'); const headerTitleClone = headerTitleElem.cloneNode(true); // Remove the original header title element headerTitleElem.remove(); // Add the hello text node and the cloned header title element to the new span element spanElem.appendChild(helloText); spanElem.appendChild(headerTitleClone); // Insert the new span element before the item lister const itemListerElem = document.querySelector('#main-header'); parentElem.insertBefore(spanElem, itemListerElem);
